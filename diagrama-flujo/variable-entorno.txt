tabla Usuario:
CREATE TABLE `dungeon_and_dragons`.`usuarios` (`USU_id` INT NOT NULL AUTO_INCREMENT , `USU_nombre` VARCHAR(45) NOT NULL , `USU_pass` VARCHAR(45) NOT NULL , `USU_correo` VARCHAR(45) NOT NULL , `USU_JUG_id` INT NOT NULL , PRIMARY KEY (`USU_id`)) ENGINE = InnoDB;

Tabla Jugadores:
CREATE TABLE `dungeon_and_dragons`.`jugadores` (`JUG_id` INT NOT NULL AUTO_INCREMENT , `JUG_nombre` VARCHAR(45) NOT NULL , `JUG_nivel` INT NOT NULL , `JUG_salud` INT NOT NULL , `JUG_raza` VARCHAR(45) NOT NULL , `JUG_experiencia` INT NOT NULL , `JUG_puntaje` INT NOT NULL , `JUG_INV_id` INT NOT NULL , PRIMARY KEY (`JUG_id`)) ENGINE = InnoDB;

Tabla Inventarios:
CREATE TABLE `dungeon_and_dragons`.`inventarios` (`INV_id` INT NOT NULL AUTO_INCREMENT , `INV_ARM_id` INT NOT NULL , `INV_ESC_id` INT NOT NULL , `INV_POC_id` INT NOT NULL , `INV_capacidad` INT NOT NULL , PRIMARY KEY (`INV_id`)) ENGINE = InnoDB;

Tabla Armas:
CREATE TABLE `dungeon_and_dragons`.`armas` (`ARM_id` INT NOT NULL AUTO_INCREMENT , `ARM_nombre` VARCHAR(45) NOT NULL , `ARM_danho` INT NOT NULL , `ARM_nivel` INT NOT NULL , PRIMARY KEY (`ARM_id`)) ENGINE = InnoDB;

Tabla Escudos:
CREATE TABLE `dungeon_and_dragons`.`escudos` (`ESC_id` INT NOT NULL AUTO_INCREMENT , `ESC_nombre` VARCHAR(45) NOT NULL , `ESC_defensa` INT NOT NULL , PRIMARY KEY (`ESC_id`)) ENGINE = InnoDB;

Tabla Pociones:
CREATE TABLE `dungeon_and_dragons`.`pociones` (`POC_id` INT NOT NULL AUTO_INCREMENT , `POC_nombre` VARCHAR(45) NOT NULL , `POC_cura` INT NOT NULL , PRIMARY KEY (`POC_id`)) ENGINE = InnoDB;

Tabla Equipos:
CREATE TABLE `dungeon_and_dragons`.`equipos` (`EQU_id` INT NOT NULL AUTO_INCREMENT , `EQU_nombre` VARCHAR(45) NOT NULL , `EQU_JUG_id` INT NOT NULL , PRIMARY KEY (`EQU_id`)) ENGINE = InnoDB;

Tabla Ranking:
CREATE TABLE `dungeon_and_dragons`.`ranking` (`RG_id` INT NOT NULL AUTO_INCREMENT , `RG_JUG_id` INT NOT NULL , PRIMARY KEY (`RG_id`)) ENGINE = InnoDB;
